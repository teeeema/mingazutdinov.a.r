LOAD 365       # Загрузка числа 365 в аккумулятор
WRITE 0        # Запись значения аккумулятора в память по адресу 0
LOAD -123      # Загрузка числа -123 в аккумулятор
ABS 1          # Запись абсолютного значения аккумулятора в память по адресу 1
READ           # Чтение значения из памяти по адресу, равному аккумулятору
WRITE 2        # Запись значения аккумулятора в память по адресу 2
